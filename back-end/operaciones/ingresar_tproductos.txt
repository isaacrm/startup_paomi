<?php
	require_once("conexion.php");
	$conexion = conectaDB(); //establecer conexion en base de datos
	$nombre = $_POST['nombre'];
	$desc = $_POST['desc'];

	$sql_query->bindParam(2, $desc);

	{
		header("location: consulta_tproducto.php");
	}
	
?>
